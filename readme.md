```python
>>> t1 = (1,2)
>>> t2 = (3,4)
>>> 
>>> l = []
>>> l.append(t1)
>>> l.append(t2)
>>> 
>>> l
[(1, 2), (3, 4)]
>>> 
>>> d = dict()
>>> 
>>> d[0]=l
>>> 
>>> d
{0: [(1, 2), (3, 4)]}
>>> 
>>> d[0]
[(1, 2), (3, 4)]
>>> d[0][1]
(3, 4)
>>> d[0][1][0]
3
>>> d[0][1][1]
4
>>> 
```
